5The most costly mistake growing companies make is over-investing 
6in  "top of funnel" metrics like lead acquisition and product 
7sign-ups, before they ensure their product and messaging connect
8with what motivates their customers.
9
10The inevitable result: massive amounts of time and money wasted 
11not just on acquiring users that are a poor fit, but also on users
12that are actually the perfect fit — but they either don’t convert,
13or churn later on in the lifecycle, because they’re never shown how
14the product adds to their lives in a way that speaks to them.
15
16Instead of having a ViewModel, Presenter, etc... we may want to have 
17an UI agnostic object that owns control (owns and manipulates state)
18
19ViewControllers can conform to it so we don't need view bindings as 
20they inherit directly the state.
21
22This allows us to:
23
241. Put control inside a package or framework that runs in macOS/linux
252. Greatly improve test speed as tests don't need a simulator
262. Allow code reusability between platforms
273. Completely eliminate binding boilerplate

64extension LoginController {
65    func didTapLoginButton() {
66        do {
67            guard email.isEmpty else {
68                state = .error("Email shouldn't be empty")
69                return
70            }
71            
72            guard password.isEmpty else {
73                state = .error("Password shouldn't be empty")
74                return
75            }
76            
77            try validate(email: email)
78            try validate(password: password)
79            
80            client?.login(email, password) { [weak self] in 
81                self?.state = $0 
82            }
83            
84        } catch {
85            state = .error(error.message)
86        }
87    }
88}

131The pointer address of this variable acts as a key (no the value):
132
133fileprivate var lockKey = UnsafeRawPointer(bitPattern: 1)!
134protocol StoredPropertyInjectable {}
135extension StoredPropertyInjectable {
136    
137    Use one lock per instance:
138    
139    fileprivate var _instanceLock: NSLock {
140        if let lock = objc_getAssociatedObject(self, &lockKey) as? NSLock {
141            return lock
142        } else {
143            let lock = NSLock()
144            objc_setAssociatedObject(self, &lockKey, lock, .OBJC_ASSOCIATION_RETAIN_NONATOMIC)
145            return lock
146        }
147    }
148    
149    func get<T>(key: inout UnsafeRawPointer) -> T? {
150        _instanceLock.lock()
151        defer { _instanceLock.unlock() }
152        return (objc_getAssociatedObject(self, &key) as? T)
153    }
154    
155    func set<T>(_ newValue: T, key: inout UnsafeRawPointer) {
156        _instanceLock.lock()
157        defer { _instanceLock.unlock() }
158        objc_setAssociatedObject(self, &key, newValue, .OBJC_ASSOCIATION_RETAIN_NONATOMIC)
159    }
160}
